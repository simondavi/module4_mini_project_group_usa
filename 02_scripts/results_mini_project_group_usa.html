# We use the analysis-ready dataset (ds_final, N = 69,534), including post-stratification weights: https://osf.io/5c3qd/wiki?wiki=fqmdn
data <- read.csv(here("00_data/ds_final.csv"), sep = ";") %>%
  filter(COUNTRY_CODE == "USA")

Compute Indices

####  ------------ Compute Indices ------------ ####

# SCIPOP_goertz: Science-related populist attitudes

# 8 items on 4 dimensions, measured with 5-point scales (1 = strongly disagree to 5 = strongly agree; Mede, 
# Schäfer, & Füchslin, 2021), aggregated with the Goertz approach (Wuttke et al., 2020)

# Compute overall and dimension scores

data <- data %>%
  mutate(
    SCIPOP_PPL_m = rowMeans(across(c(SCIPOP_common, SCIPOP_good)), na.rm = TRUE),
    SCIPOP_ELI_m = rowMeans(across(c(SCIPOP_advantage, SCIPOP_cahoots)), na.rm = TRUE),
    SCIPOP_DEC_m = rowMeans(across(c(SCIPOP_influence, SCIPOP_involved)), na.rm = TRUE),
    SCIPOP_TRU_m = rowMeans(across(c(SCIPOP_lifeexp, SCIPOP_rely)), na.rm = TRUE),
    SCIPOP_goertz = pmin(SCIPOP_PPL_m, SCIPOP_ELI_m, SCIPOP_DEC_m, SCIPOP_TRU_m, na.rm = TRUE)
  )

data <- labelled::set_variable_labels(
  data,
  SCIPOP_goertz = "Science-related populist attitudes: overall (Goertz score)",
  SCIPOP_PPL_m = "Science-related populist attitudes: conceptions of the ordinary people (mean)",
  SCIPOP_ELI_m = "Science-related populist attitudes: conceptions of the academic elite (mean)",
  SCIPOP_DEC_m = "Science-related populist attitudes: demands for decision-making sovereignty (mean)",
  SCIPOP_TRU_m = "Science-related populist attitudes: demands for truth-speaking sovereignty (mean)"
)

# Gender
data <- data %>%
  mutate(DEM_GENDER = as.factor(DEM_GENDER))
